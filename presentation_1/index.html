Fibonacci
int fibDynamic(int i) {
    i < 2 ? 1 : fibDynamic(i - 2) + fibDynamic(i - 1)
}

@CompileStatic
int fibStatic(int i) {
    i < 2 ? 1 : fibStatic(i - 2) + fibStatic(i - 1)
}
